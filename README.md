# d3-stack-time

一个D3时间堆叠图插件。

## 用法
```
// 导入D3与本插件
var d3 = require('d3');
var chart = require('d3-stack-time');

var data = {
  axisX: ['5-1', '5-2', '5-3', '5-4', '5-5', '5-6', '5-7', '5-8', '5-9', '5-10', '5-11', '5-12', '5-13', '5-14', '5-15', '5-16'],
  values: ["", "", " 08:48 18:40", " 09:20 20:26", " 09:06 20:33", " 09:00 19:20", " 09:15 18:11", "", " 09:27 21:04", " 09:24 19:05", " 09:16 18:47", " 09:14 19:09", " 09:06 20:06", "", "", " 09:35 19:00"]
};
d3.select('.time-stack').call(chart()
  .data(data) // 数据
  .titleText('test_title') // 图表标题
  .stackLabelCustom({ // 自定义图例
    'workBefore': 'Are you Ready?'
  })
);

```
